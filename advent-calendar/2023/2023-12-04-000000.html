// #!/usr/bin/env node

const fs = require('fs');
const dreams = JSON.parse(fs.readFileSync('2023-12-19.json', 'utf8'));

dreams.forEach((dream) => {
    const titleElement = typeof dream.url === "undefined"
        ? `<span>${dream.title}</span>`
        : `<a href="${dream.url}" target="_blank">${dream.title}</a>`

    console.log(
        `<tr>
            <td><time datetime="${dream.date}">${dream.date}</time></td>
            <td>${titleElement}</td>
            <td><a href="${dream.artistUrl}" target="_blank">${dream.artistName}</a></td>
        </tr>`
    )
})
